# Todo App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a full todo app on `/todo` — top submit form (task/priority/due date), sorted list below with toggle/edit/delete, backed by Cloudflare D1 via Hono Worker, with single-source schema in `src/shared/schemas/`.

**Architecture:** Drizzle + zod schema lives in `src/shared/schemas/todo.ts` (single source of truth); `src/worker/db/schema.ts` re-exports it so `WorkerDB` typing is unchanged. Hono route `src/worker/routes/todos.ts` serves RESTful CRUD with server-side SQL `ORDER BY`. Four React components in `src/components/todo/` (form / edit dialog / delete dialog / list) are composed in `src/pages/todo.tsx`. Browser-side `src/lib/api.ts` wraps `fetch`. Time fields stored as ms timestamps (integer) in D1.

**Tech Stack:** Next.js 16 (static export, Pages Router), React 19, Hono 4, Drizzle ORM 0.45, Cloudflare D1, react-hook-form 7, zod 4, `@hookform/resolvers`, date-fns 4, react-day-picker 10, shadcn/ui (radix-nova), sonner.

```ts
import { sqliteTable, text, integer } from "drizzle-orm/sqlite-core";
import { z } from "zod";

// Local-timezone start of today as a ms timestamp.
// dueDate values submitted by the form are also local start-of-day timestamps,
// so create-time validation compares apples to apples.
export function startOfTodayMs(): number {
  const d = new Date();
  d.setHours(0, 0, 0, 0);
  return d.getTime();
}

export const priorityEnum = z.enum(["low", "medium", "high"]);

export const createTodoSchema = z.object({
  task: z.string().trim().min(1, "任务不能为空").max(200, "任务最多 200 字符"),
  priority: priorityEnum,
  dueDate: z
    .number()
    .int()
    .refine((n) => n >= startOfTodayMs(), "不能选过去日期"),
});

export const updateTodoSchema = z.object({
  task: z.string().trim().min(1, "任务不能为空").max(200, "任务最多 200 字符").optional(),
  priority: priorityEnum.optional(),
  // edit allows keeping an existing past due date (no past-date check here)
  dueDate: z.number().int().optional(),
  completed: z.boolean().optional(),
});

export const todoSchema = z.object({
  id: z.string(),
  task: z.string(),
  priority: priorityEnum,
  dueDate: z.number().int(),
  completed: z.boolean(),
  createdAt: z.number().int(),
  updatedAt: z.number().int(),
});

export const todos = sqliteTable("todos", {
  id: text("id").primaryKey(),
  task: text("task").notNull(),
  priority: text("priority").notNull().default("medium"),
  dueDate: integer("due_date").notNull(),
  completed: integer("completed").notNull().default(0),
  createdAt: integer("created_at").notNull(),
  updatedAt: integer("updated_at").notNull(),
});

export type Priority = z.infer<typeof priorityEnum>;
export type Todo = z.infer<typeof todoSchema>;
export type CreateTodoInput = z.infer<typeof createTodoSchema>;
export type UpdateTodoInput = z.infer<typeof updateTodoSchema>;
```

- [ ] **Step 1: Create `src/worker/routes/todos.ts`**

```ts
import { Hono } from "hono";
import { asc, sql, eq } from "drizzle-orm";
import type { AppEnv } from "@/worker/types";
import { BusinessError } from "@/worker/lib/errors";
import {
  todos,
  createTodoSchema,
  updateTodoSchema,
  type Todo,
  type CreateTodoInput,
  type UpdateTodoInput,
} from "@/shared/schemas";

export const todosRoute = new Hono<AppEnv>();

// D1 row (snake_case columns, completed as 0/1) → API Todo (camelCase, completed as boolean).
function toTodo(row: typeof todos.$inferSelect): Todo {
  return {
    id: row.id,
    task: row.task,
    priority: row.priority as Todo["priority"],
    dueDate: row.dueDate,
    completed: row.completed === 1,
    createdAt: row.createdAt,
    updatedAt: row.updatedAt,
  };
}

// GET /api/todos — server-side SQL ORDER BY.
// Order: uncompleted first, then high→medium→low, then earlier dueDate, then earlier createdAt.
todosRoute.get("/todos", async (c) => {
  const db = c.get("db");
  const rows = await db
    .select()
    .from(todos)
    .orderBy(
      asc(todos.completed),
      sql`CASE ${todos.priority} WHEN 'high' THEN 3 WHEN 'medium' THEN 2 ELSE 1 END DESC`,
      asc(todos.dueDate),
      asc(todos.createdAt),
    );
  return c.json(rows.map(toTodo));
});

// POST /api/todos
todosRoute.post("/todos", async (c) => {
  const parsed = createTodoSchema.safeParse(await c.req.json());
  if (!parsed.success) {
    throw new BusinessError(
      parsed.error.issues[0]?.message ?? "参数错误",
      400,
    );
  }
  const input: CreateTodoInput = parsed.data;
  const now = Date.now();
  const id = crypto.randomUUID();
  const db = c.get("db");
  const row = {
    id,
    task: input.task,
    priority: input.priority,
    dueDate: input.dueDate,
    completed: 0,
    createdAt: now,
    updatedAt: now,
  };
  await db.insert(todos).values(row);
  return c.json(toTodo({ ...row, completed: 0 }), 201);
});

// PATCH /api/todos/:id — partial update; updatedAt always refreshed.
todosRoute.patch("/todos/:id", async (c) => {
  const id = c.req.param("id");
  const parsed = updateTodoSchema.safeParse(await c.req.json());
  if (!parsed.success) {
    throw new BusinessError(
      parsed.error.issues[0]?.message ?? "参数错误",
      400,
    );
  }
  const input: UpdateTodoInput = parsed.data;
  const db = c.get("db");

  const patch: Partial<typeof todos.$inferInsert> = { updatedAt: Date.now() };
  if (input.task !== undefined) patch.task = input.task;
  if (input.priority !== undefined) patch.priority = input.priority;
  if (input.dueDate !== undefined) patch.dueDate = input.dueDate;
  if (input.completed !== undefined) patch.completed = input.completed ? 1 : 0;

  const result = await db.update(todos).set(patch).where(eq(todos.id, id));
  // D1 returns { meta: { changes } }; drizzle surfaces it on the result.
  const changes = (result as unknown as { meta?: { changes?: number } })?.meta?.changes ?? 0;
  if (changes === 0) {
    throw new BusinessError("任务不存在", 404);
  }

  const updated = await db.select().from(todos).where(eq(todos.id, id)).get();
  if (!updated) {
    throw new BusinessError("任务不存在", 404);
  }
  return c.json(toTodo(updated));
});

// DELETE /api/todos/:id
todosRoute.delete("/todos/:id", async (c) => {
  const id = c.req.param("id");
  const db = c.get("db");
  const result = await db.delete(todos).where(eq(todos.id, id));
  const changes = (result as unknown as { meta?: { changes?: number } })?.meta?.changes ?? 0;
  if (changes === 0) {
    throw new BusinessError("任务不存在", 404);
  }
  return c.body(null, 204);
});
```

```ts
import type {
  Todo,
  CreateTodoInput,
  UpdateTodoInput,
} from "@/shared/schemas";

export class ApiError extends Error {
  code: number;
  constructor(message: string, code: number) {
    super(message);
    this.name = "ApiError";
    this.code = code;
  }
}

async function parseError(res: Response): Promise<ApiError> {
  try {
    const body = (await res.json()) as { code?: number; message?: string; data?: unknown };
    return new ApiError(body.message ?? "请求失败", body.code ?? res.status);
  } catch {
    return new ApiError("请求失败", res.status);
  }
}

export async function listTodos(): Promise<Todo[]> {
  const res = await fetch("/api/todos");
  if (!res.ok) throw await parseError(res);
  return (await res.json()) as Todo[];
}

export async function createTodo(input: CreateTodoInput): Promise<Todo> {
  const res = await fetch("/api/todos", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(input),
  });
  if (!res.ok) throw await parseError(res);
  return (await res.json()) as Todo;
}

export async function updateTodo(
  id: string,
  input: UpdateTodoInput,
): Promise<Todo> {
  const res = await fetch(`/api/todos/${id}`, {
    method: "PATCH",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(input),
  });
  if (!res.ok) throw await parseError(res);
  return (await res.json()) as Todo;
}

export async function deleteTodo(id: string): Promise<void> {
  const res = await fetch(`/api/todos/${id}`, { method: "DELETE" });
  if (!res.ok) throw await parseError(res);
  // 204 no body
}
```

import * as React from "react";
import { useForm, Controller } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { format } from "date-fns";
import { CalendarIcon } from "lucide-react";

import {
  createTodoSchema,
  updateTodoSchema,
  startOfTodayMs,
  type CreateTodoInput,
  type UpdateTodoInput,
  type Priority,
} from "@/shared/schemas";
import { cn } from "@/lib/utils";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import {
  Field,
  FieldLabel,
  FieldError,
  FieldDescription,
} from "@/components/ui/field";
import { Popover, PopoverContent, PopoverTrigger } from "@/components/ui/popover";
import { Calendar } from "@/components/ui/calendar";
import { Spinner } from "@/components/ui/spinner";
import { Select } from "@/components/select";

type FormValues = {
  task: string;
  priority: Priority;
  dueDate: number;
};

type TodoFormProps = {
  defaultValues?: { task: string; priority: Priority; dueDate: number };
  onSubmit: (input: CreateTodoInput | UpdateTodoInput) => Promise<void>;
  submitLabel: string;
  onCancel?: () => void;
  submitting?: boolean;
};

const PRIORITY_OPTIONS = [
  { value: "low", label: "低" },
  { value: "medium", label: "中" },
  { value: "high", label: "高" },
];

export function TodoForm({
  defaultValues,
  onSubmit,
  submitLabel,
  onCancel,
  submitting,
}: TodoFormProps) {
  const isEdit = defaultValues !== undefined;
  const resolver = zodResolver(isEdit ? updateTodoSchema : createTodoSchema);

  const form = useForm<FormValues>({
    resolver: resolver as never,
    defaultValues: defaultValues ?? {
      task: "",
      priority: "medium",
      dueDate: startOfTodayMs(),
    },
  });

  async function handleSubmit(values: FormValues) {
    // For edit mode, send only the fields; updateTodoSchema accepts them as optional.
    // For create mode, send all three.
    await onSubmit(values as CreateTodoInput | UpdateTodoInput);
  }

  return (

                    nd.setHours(0, 0, 0, 0);

            </Popover>

              <FieldDescription>不能选择今天之前的日期</FieldDescription>
            )}
            {fieldState.invalid && <FieldError errors={[fieldState.error]} />}
          </Field>
        )}

          <Button type="button" variant="ghost" onClick={onCancel} disabled={submitting}>
            取消
          </Button>

```tsx
"use client";

import * as React from "react";
import { toast } from "sonner";

import { Page } from "@/components/layout/page";
import { TodoForm } from "@/components/todo/todo-form";
import { TodoEditDialog } from "@/components/todo/todo-edit-dialog";
import { TodoDeleteDialog } from "@/components/todo/todo-delete-dialog";
import { TodoList } from "@/components/todo/todo-list";
import {
  listTodos,
  createTodo,
  updateTodo,
  deleteTodo,
  ApiError,
} from "@/lib/api";
import type { Todo, CreateTodoInput, UpdateTodoInput } from "@/shared/schemas";

export default function TodoPage() {
  const [todos, setTodos] = React.useState<Todo[]>([]);
  const [loading, setLoading] = React.useState(true);
  const [error, setError] = React.useState<string | null>(null);
  const [creating, setCreating] = React.useState(false);
  const [editingTodo, setEditingTodo] = React.useState<Todo | null>(null);
  const [deletingTodo, setDeletingTodo] = React.useState<Todo | null>(null);

  const refresh = React.useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      setTodos(await listTodos());
    } catch (e) {
      const msg = e instanceof ApiError ? e.message : "加载失败";
      setError(msg);
    } finally {
      setLoading(false);
    }
  }, []);

  React.useEffect(() => {
    void refresh();
  }, [refresh]);

  async function handleCreate(input: CreateTodoInput) {
    setCreating(true);
    try {
      await createTodo(input);
      await refresh();
      toast.success("已添加");
    } catch (e) {
      toast.error(e instanceof ApiError ? e.message : "添加失败");
    } finally {
      setCreating(false);
    }
  }

  async function handleToggle(t: Todo) {
    try {
      await updateTodo(t.id, { completed: !t.completed });
      await refresh();
      toast.success(t.completed ? "已取消完成" : "已完成");
    } catch (e) {
      toast.error(e instanceof ApiError ? e.message : "操作失败");
    }
  }

  // Passed to TodoEditDialog as onSubmit. The dialog delegates to this;
  // the page owns the API call, toast, and dialog closing.
  async function handleUpdate(input: UpdateTodoInput) {
    if (!editingTodo) return;
    try {
      await updateTodo(editingTodo.id, input);
      setEditingTodo(null);
      await refresh();
      toast.success("已保存");
    } catch (e) {
      toast.error(e instanceof ApiError ? e.message : "保存失败");
    }
  }

  async function handleDelete() {
    if (!deletingTodo) return;
    try {
      await deleteTodo(deletingTodo.id);
      setDeletingTodo(null);
      await refresh();
      toast.success("已删除");
    } catch (e) {
      toast.error(e instanceof ApiError ? e.message : "删除失败");
    }
  }

  return (
    <Page>
      <div className="mx-auto flex w-full max-w-2xl flex-col gap-6 py-8">
        <header>
          <h1 className="font-heading text-[28px] font-semibold tracking-tight">
            Todo
          </h1>
        </header>

        <section className="rounded-lg border border-border bg-card p-4">
          <TodoForm
            onSubmit={handleCreate}
            submitLabel="添加"
            submitting={creating}
          />
        </section>

        <section>
          <TodoList
            todos={todos}
            loading={loading}
            error={error}
            onToggle={handleToggle}
            onEdit={setEditingTodo}
            onDelete={setDeletingTodo}
            onRetry={refresh}
          />
        </section>
      </div>

      {editingTodo && (
        <TodoEditDialog
          todo={editingTodo}
          onSubmit={handleUpdate}
          onOpenChange={(open) => {
            if (!open) setEditingTodo(null);
          }}
        />
      )}
      {deletingTodo && (
        <TodoDeleteDialog
          onConfirm={handleDelete}
          onOpenChange={(open) => {
            if (!open) setDeletingTodo(null);
          }}
        />
      )}
    </Page>
  );
}
```
